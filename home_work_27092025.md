### **–ü—Ä–∏–º–µ—Ä 5: –ü—Ä–æ–∏–∑–≤–æ–¥–∏—Ç–µ–ª—å-–ü–æ—Ç—Ä–µ–±–∏—Ç–µ–ª—å —Å Queue**
```python
import threading
import queue
import time
import random

def producer(q, name):
    """–ü—Ä–æ–∏–∑–≤–æ–¥–∏—Ç –¥–∞–Ω–Ω—ã–µ –∏ –∫–ª–∞–¥–µ—Ç –≤ –æ—á–µ—Ä–µ–¥—å"""
    for i in range(5):
        item = f"–ü—Ä–æ–¥—É–∫—Ç –æ—Ç {name}-{i}"
        q.put(item)
        print(f"üõ†Ô∏è  {name} –ø—Ä–æ–∏–∑–≤–µ–ª: {item}")
        time.sleep(random.uniform(0.1, 0.5))
    q.put(None)  # –°–∏–≥–Ω–∞–ª –∑–∞–≤–µ—Ä—à–µ–Ω–∏—è

def consumer(q, name):
    """–ü–æ—Ç—Ä–µ–±–ª—è–µ—Ç –¥–∞–Ω–Ω—ã–µ –∏–∑ –æ—á–µ—Ä–µ–¥–∏"""
    while True:
        item = q.get()
        if item is None:
            q.put(None)  # –ü–µ—Ä–µ–¥–∞–µ–º —Å–∏–≥–Ω–∞–ª –¥–∞–ª—å—à–µ
            break
        print(f"üõí {name} –∫—É–ø–∏–ª: {item}")
        time.sleep(random.uniform(0.2, 0.7))
        q.task_done()

# –°–æ–∑–¥–∞–µ–º –æ—á–µ—Ä–µ–¥—å
q = queue.Queue()

# –°–æ–∑–¥–∞–µ–º –ø–æ—Ç–æ–∫–∏

    threading.Thread(target=producer, args=(q, "–ó–∞–≤–æ–¥-A")),
    threading.Thread(target=producer, args=(q, "–ó–∞–≤–æ–¥-B"))
]

consumers = [
    threading.Thread(target=consumer, args=(q, "–ú–∞–≥–∞–∑–∏–Ω-1")),
    threading.Thread(target=consumer, args=(q, "–ú–∞–≥–∞–∑–∏–Ω-2"))
]

# –ó–∞–ø—É—Å–∫–∞–µ–º –≤—Å–µ—Ö
for p in producers:
    p.start()

for c in consumers:
    c.start()

# –ñ–¥–µ–º –∑–∞–≤–µ—Ä—à–µ–Ω–∏—è –ø—Ä–æ–∏–∑–≤–æ–¥–∏—Ç–µ–ª–µ–π
```
